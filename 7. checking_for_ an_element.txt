def is_anyone_in(collection, city):  # создаем функцию
    if city in collection.values():  # если есть среди значений словаря collection 
        for name in collection.keys():  # переберите все ключи словаря
            if collection[name] == city:  # если соответствующее ключу значение равно city
                print('В городе ' + city + ' живёт ' + name + '.')  # выводим строку с информацией 
    else:  # иначе
        print('Пока никого.')  # ничего не выводим 

# заполненый словарь 
friends = {               
    'Серёга': 'Омск', 
    'Соня': 'Москва', 
    'Дима': 'Челябинск', 
    'Алина': 'Хабаровск', 
    'Егор': 'Пермь'
}

is_anyone_in(friends, 'Хабаровск')  # вызов функции и указывааем критерии перебора

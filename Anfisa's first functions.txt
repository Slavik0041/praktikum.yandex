# словарь друзей и городов в которых они проживают 
DATABASE = {
    'Серёга': 'Омск',
    'Соня': 'Москва',
    'Миша': 'Москва',
    'Дима': 'Челябинск',
    'Алина': 'Красноярск',
    'Егор': 'Пермь',
    'Коля': 'Красноярск'
}


# 3-я ф-ция
# функциф которая опр. окончание ответа на вопрос сколько у тебя друзей
def format_friends(friends_count):  # объявляем функцию с условием/переменной friends_count
    if friends_count == 1: # если 1 друг то возвразщаем 'У тебя N друг'
        return 'У тебя 1 друг'  # возращаем 'У тебя N друг'
    elif 2 <= friends_count <= 4:  # если от 2х до 4х друзей возвращаем 'У тебя N друга'
        return 'У тебя ' + str(friends_count) + ' друга'  # возвращаем 'У тебя N друга'
    elif friends_count >= 5:   # если 5 и больше возвращаем 'У тебя N друзей'
        return 'У тебя ' + str(friends_count) + ' друзей'   # возвращаем 'У тебя N друзей'


# 2-я ф-ция
# функция отвечающая на вопросы пользовател 
def process_query(query):   # объявление фнкции с параметром query
    if query == 'Сколько у меня друзей?':   # если вопрос, то 
        count = len(DATABASE)   # считаем кол-во записей в массиве 
        return format_friends(count)   # взвращаем функцияю format_friends с получившимся числом 
    elif query == 'Кто все мои друзья?':   # если вопрос, то 
        friends_string = ', '.join(DATABASE)   # объявляем множество, разделяем базу друзей по запятым и вносим в множество  
        return 'Твои друзья: ' + friends_string   # возвращаем ответ 'Твои друзья ' и множество (друзей)
    else:   # в остальных случаях 
        return '<неизвестный запрос>'   # возвращаем неизвестный запрос


# 1-я ф-ция
#  функция с перечнем вопросов с которыми будем работать 
def runner():   # объвление функции 
    print('Привет, я Анфиса!') # приветствие 
    print(process_query('Сколько у меня друзей?'))  # обявляем ф-цию и передаем в нее вопрос 
    print(process_query('Кто все мои друзья?'))   # обявляем ф-цию и передаем в нее вопрос
    print(process_query('Где все мои друзья?'))   # обявляем ф-цию и передаем в нее вопрос


runner()   # объявлеям 1ю ф-цию и дальше по цепочке

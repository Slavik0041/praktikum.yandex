# функция объеденения 2х словарей во множество 
def print_shopping_list(pizza, salad): # объявление функции
    spisok = set(pizza) # преобразуем словарь во множестов 
    spisok = spisok.union(salad) # добавим во множество список ингридиентов 2го блюда 
    for i in spisok:  # переберем по наличию ключа в словарях, при наличии суммируем значения
        ves = 0 # иницилизируем переменную в которой будем хранить значение ключа
        if i in pizza.keys(): ves += pizza[i] # прошли по 1му словарю ингридиентов
        if i in salad.keys(): ves += salad[i] # прошли по 2му словарю ингридиентов 
        print(i, ": ", ves) # выводим результат 

# словарь продуктов 1го блюда         
pizza = {'мука, кг': 1,
         'помидоры, кг': 1.5,
         'шампиньоны, кг': 1.5,
         'сыр, кг': 0.8,
         'оливковое масло, л': 0.1,
         'дрожжи, г': 50}

# словарь продуктов 2го блюда 
salad = {'огурцы, кг': 1,
         'перцы, кг': 1,
         'помидоры, кг': 1.5,
         'оливковое масло, л': 0.1,
         'листья салата, кг': 0.4}

print_shopping_list(pizza, salad) # вызоф функции
